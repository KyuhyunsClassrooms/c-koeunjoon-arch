## 3. 데이터 구조 설계
(프로그램에 필요한 핵심 변수, 배열을 설계하세요. 필요하면 구조체(struct)를 사용할 수도 있습니다.)

**방법 1: 배열 사용 (기본)**
```c
// 예시: 학생 5명의 이름과 성적 저장
char names[5][20];
int scores[5];
int count = 0;  // 현재 학생 수
```
// 책 100권의 제목, 저자, 가격을 각각 배열로 저장
char titles[100][100];   // 책 제목
char authors[100][100];  // 저자 이름
int prices[100];         // 가격

int count = 0;  // 현재 등록된 책 수

**방법 2: 구조체 사용 (선택)**
```c
// 예시: 학생 정보를 구조체로 묶어서 관리
struct Student {
    char name[20];
    int kor;
    int eng;
    int math;
    float avg;
};
struct Student students[5];  // 학생 5명 저장
```

// 책 정보를 하나의 구조체로 묶어서 관리
struct Book {
    char title[100];    // 책 제목
    char author[100];   // 저자 이름
    int price;          // 가격
};

// 책 100권 저장 가능
struct Book books[100];

int count = 0;   // 현재 등록된 책 수
